// Message.java
import java.util.*;
import java.io.*;
import org.json.JSONObject;
import org.json.JSONArray;

public class Message {
    private String messageID;
    private int numMessagesSent;
    private String recipientNumber;
    private String message;
    private String messageHash;
    private static int messageCounter = 0;
    private static List<Message> sentMessages = new ArrayList<>();
    private static List<Message> storedMessages = new ArrayList<>();
    private static final String STORAGE_FILE = "messages.json";
    
    // Constructor
    public Message(String recipientNumber, String message) {
        this.recipientNumber = recipientNumber;
        this.message = message;
        this.messageID = generateMessageID();
        this.numMessagesSent = ++messageCounter;
        this.messageHash = createMessageHash();
    }
    
    // Generate random 10-digit message ID
    private String generateMessageID() {
        Random rand = new Random();
        StringBuilder id = new StringBuilder();
        for (int i = 0; i < 10; i++) {
            id.append(rand.nextInt(10));
        }
        return id.toString();
    }
    
    // Method 1: Check if message ID is not more than 10 characters
    public boolean checkMessageID() {
        return messageID != null && messageID.length() <= 10;
    }
    
    // Method 2: Check recipient cell number format
    public String checkRecipientCell() {
        if (recipientNumber == null || recipientNumber.isEmpty()) {
            return "Cell phone number is incorrectly formatted or does not contain an international code. Please correct the number and try again.";
        }
        
        // Check for international code (+27 for SA, or any +XX format)
        boolean hasInternationalCode = recipientNumber.startsWith("+");
        
        // Check length (international numbers are 10-15 digits including +)
        boolean validLength = recipientNumber.length() >= 10 && recipientNumber.length() <= 15;
        
        // Check that after the +, all characters are digits
        boolean digitsOnly = true;
        for (int i = 1; i < recipientNumber.length(); i++) {
            char c = recipientNumber.charAt(i);
            if (c < '0' || c > '9') {
                digitsOnly = false;
                break;
            }
        }
        
        if (hasInternationalCode && validLength && digitsOnly) {
            return "Cell phone number successfully captured.";
        } else {
            return "Cell phone number is incorrectly formatted or does not contain an international code. Please correct the number and try again.";
        }
    }
    
    // Method 3: Create message hash
    public String createMessageHash() {
        // Format: first two numbers of message ID:message number:first and last words in caps
        String firstTwoID = messageID.substring(0, 2);
        String[] words = message.split(" ");
        
        String firstWord = words.length > 0 ? words[0] : "";
        String lastWord = words.length > 1 ? words[words.length - 1] : firstWord;
        
        // Remove any punctuation from words for cleaner hash
        firstWord = firstWord.replaceAll("[^a-zA-Z]", "");
        lastWord = lastWord.replaceAll("[^a-zA-Z]", "");
        
        String hash = firstTwoID + ":" + numMessagesSent + ":" + firstWord + lastWord;
        return hash.toUpperCase();
    }
    
    // Method 4: Send/Store/Disregard message
    public String sentMessage(int choice) {
        switch (choice) {
            case 1: // Send Message
                return sendMessage();
                
            case 2: // Store Message
                return storeMessage();
                
            case 0: // Disregard/Delete Message
                return "Press 0 to delete the message.";
                
            default:
                return "Invalid option.";
        }
    }
    
    private String sendMessage() {
        sentMessages.add(this);
        saveToJSON();
        return "Message successfully sent.\nPress 0 to delete the message.\nMessage successfully stored.";
    }
    
    private String storeMessage() {
        storedMessages.add(this);
        saveToJSON();
        return "Message successfully stored.";
    }
    
    // Method 5: Print all messages
    public String printMessages() {
        if (sentMessages.isEmpty() && storedMessages.isEmpty()) {
            return "No messages sent or stored yet.";
        }
        
        StringBuilder output = new StringBuilder();
        output.append("=== SENT MESSAGES ===\n");
        if (sentMessages.isEmpty()) {
            output.append("No sent messages.\n");
        } else {
            for (int i = 0; i < sentMessages.size(); i++) {
                Message msg = sentMessages.get(i);
                output.append((i + 1)).append(". ");
                output.append("ID: ").append(msg.messageID).append(" | ");
                output.append("To: ").append(msg.recipientNumber).append(" | ");
                output.append("Hash: ").append(msg.messageHash).append(" | ");
                output.append("Message: ").append(msg.message).append("\n");
            }
        }
        
        output.append("\n=== STORED MESSAGES ===\n");
        if (storedMessages.isEmpty()) {
            output.append("No stored messages.\n");
        } else {
            for (int i = 0; i < storedMessages.size(); i++) {
                Message msg = storedMessages.get(i);
                output.append((i + 1)).append(". ");
                output.append("ID: ").append(msg.messageID).append(" | ");
                output.append("To: ").append(msg.recipientNumber).append(" | ");
                output.append("Hash: ").append(msg.messageHash).append(" | ");
                output.append("Message: ").append(msg.message).append("\n");
            }
        }
        
        return output.toString();
    }
    
    // Get total number of messages sent
    public int getTotalMessagesSent() {
        return sentMessages.size();
    }
    
    // Get total number of messages stored
    public int getTotalMessagesStored() {
        return storedMessages.size();
    }
    
    // Validate message length (max 250 characters)
    public static String validateMessageLength(String message) {
        if (message == null || message.isEmpty()) {
            return "Message ready to send.";
        }
        
        if (message.length() <= 250) {
            return "Message ready to send.";
        } else {
            int excess = message.length() - 250;
            return "Message exceeds 250 characters by " + excess + "; please reduce the size.";
        }
    }
    
    // Save messages to JSON file
    private void saveToJSON() {
        JSONObject storage = new JSONObject();
        JSONArray sentArray = new JSONArray();
        JSONArray storedArray = new JSONArray();
        
        // Load existing data if file exists
        try {
            File file = new File(STORAGE_FILE);
            if (file.exists()) {
                String content = new String(java.nio.file.Files.readAllBytes(file.toPath()));
                storage = new JSONObject(content);
                if (storage.has("sentMessages")) {
                    sentArray = storage.getJSONArray("sentMessages");
                }
                if (storage.has("storedMessages")) {
                    storedArray = storage.getJSONArray("storedMessages");
                }
            }
        } catch (Exception e) {
            // File doesn't exist or is empty, start fresh
        }
        
        // Clear and rebuild sent messages array
        sentArray = new JSONArray();
        for (Message msg : sentMessages) {
            sentArray.put(msg.toJSON());
        }
        
        // Clear and rebuild stored messages array
        storedArray = new JSONArray();
        for (Message msg : storedMessages) {
            storedArray.put(msg.toJSON());
        }
        
        storage.put("sentMessages", sentArray);
        storage.put("storedMessages", storedArray);
        storage.put("messageCounter", messageCounter);
        
        // Save to file
        try (FileWriter writer = new FileWriter(STORAGE_FILE)) {
            writer.write(storage.toString(4));
        } catch (Exception e) {
            System.out.println("Error saving messages: " + e.getMessage());
        }
    }
    
    // Convert message to JSON object
    private JSONObject toJSON() {
        JSONObject obj = new JSONObject();
        obj.put("messageID", messageID);
        obj.put("numMessagesSent", numMessagesSent);
        obj.put("recipientNumber", recipientNumber);
        obj.put("message", message);
        obj.put("messageHash", messageHash);
        obj.put("timestamp", new Date().toString());
        return obj;
    }
    
    // Load messages from JSON file
    public static void loadMessagesFromFile() {
        try {
            File file = new File(STORAGE_FILE);
            if (file.exists()) {
                String content = new String(java.nio.file.Files.readAllBytes(file.toPath()));
                JSONObject storage = new JSONObject(content);
                
                sentMessages.clear();
                storedMessages.clear();
                
                if (storage.has("sentMessages")) {
                    JSONArray sentArray = storage.getJSONArray("sentMessages");
                    for (int i = 0; i < sentArray.length(); i++) {
                        JSONObject obj = sentArray.getJSONObject(i);
                        Message msg = new Message(
                            obj.getString("recipientNumber"),
                            obj.getString("message")
                        );
                        msg.messageID = obj.getString("messageID");
                        msg.numMessagesSent = obj.getInt("numMessagesSent");
                        msg.messageHash = obj.getString("messageHash");
                        sentMessages.add(msg);
                    }
                }
                
                if (storage.has("storedMessages")) {
                    JSONArray storedArray = storage.getJSONArray("storedMessages");
                    for (int i = 0; i < storedArray.length(); i++) {
                        JSONObject obj = storedArray.getJSONObject(i);
                        Message msg = new Message(
                            obj.getString("recipientNumber"),
                            obj.getString("message")
                        );
                        msg.messageID = obj.getString("messageID");
                        msg.numMessagesSent = obj.getInt("numMessagesSent");
                        msg.messageHash = obj.getString("messageHash");
                        storedMessages.add(msg);
                    }
                }
                
                if (storage.has("messageCounter")) {
                    messageCounter = storage.getInt("messageCounter");
                }
            }
        } catch (Exception e) {
            System.out.println("No saved messages found or error loading: " + e.getMessage());
        }
    }
    
    // Delete a message by ID
    public static boolean deleteMessage(String messageID) {
        // Check sent messages
        for (int i = 0; i < sentMessages.size(); i++) {
            if (sentMessages.get(i).messageID.equals(messageID)) {
                sentMessages.remove(i);
                saveAllMessages();
                return true;
            }
        }
        
        // Check stored messages
        for (int i = 0; i < storedMessages.size(); i++) {
            if (storedMessages.get(i).messageID.equals(messageID)) {
                storedMessages.remove(i);
                saveAllMessages();
                return true;
            }
        }
        
        return false;
    }
    
    private static void saveAllMessages() {
        JSONObject storage = new JSONObject();
        JSONArray sentArray = new JSONArray();
        JSONArray storedArray = new JSONArray();
        
        for (Message msg : sentMessages) {
            sentArray.put(msg.toJSON());
        }
        
        for (Message msg : storedMessages) {
            storedArray.put(msg.toJSON());
        }
        
        storage.put("sentMessages", sentArray);
        storage.put("storedMessages", storedArray);
        storage.put("messageCounter", messageCounter);
        
        try (FileWriter writer = new FileWriter(STORAGE_FILE)) {
            writer.write(storage.toString(4));
        } catch (Exception e) {
            System.out.println("Error saving messages: " + e.getMessage());
        }
    }
    
    // Getters
    public String getMessageID() { return messageID; }
    public int getNumMessagesSent() { return numMessagesSent; }
    public String getRecipientNumber() { return recipientNumber; }
    public String getMessage() { return message; }
    public String getMessageHash() { return messageHash; }
    
    // Reset for testing purposes
    public static void resetForTesting() {
        sentMessages.clear();
        storedMessages.clear();
        messageCounter = 0;
    }
}
