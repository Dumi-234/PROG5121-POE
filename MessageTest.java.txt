// MessageTest.java
import org.junit.jupiter.api.*;
import static org.junit.jupiter.api.Assertions.*;

@TestMethodOrder(MethodOrderer.OrderAnnotation.class)
public class MessageTest {
    
    private Message message;
    
    @BeforeEach
    void setUp() {
        Message.resetForTesting();
        message = new Message("+27718693002", "Hi Mike, can you join us for dinner tonight?");
    }
    
    @AfterEach
    void tearDown() {
        Message.resetForTesting();
    }
    
    // ==================== TEST 1: Message ID Validation ====================
    
    @Test
    @Order(1)
    void testMessageIDIsNotMoreThanTenCharacters() {
        assertTrue(message.checkMessageID(), "Message ID should be 10 characters or less");
        assertTrue(message.getMessageID().length() <= 10, 
                  "Message ID length is " + message.getMessageID().length() + " which exceeds 10");
    }
    
    @Test
    @Order(2)
    void testMessageIDIsExactlyTenDigits() {
        assertEquals(10, message.getMessageID().length(), 
                    "Message ID should be exactly 10 digits long");
        
        // Verify all characters are digits
        for (char c : message.getMessageID().toCharArray()) {
            assertTrue(Character.isDigit(c), "Message ID should only contain digits");
        }
    }
    
    // ==================== TEST 2: Recipient Cell Number Validation ====================
    
    @Test
    @Order(3)
    void testValidSouthAfricanCellNumberSuccess() {
        Message validMsg = new Message("+27718693002", "Test message");
        String result = validMsg.checkRecipientCell();
        assertEquals("Cell phone number successfully captured.", result);
    }
    
    @Test
    @Order(4)
    void testValidInternationalCellNumberSuccess() {
        Message validMsg = new Message("+447911123456", "Test message");
        String result = validMsg.checkRecipientCell();
        assertEquals("Cell phone number successfully captured.", result);
    }
    
    @Test
    @Order(5)
    void testInvalidCellNumberNoInternationalCode() {
        Message invalidMsg = new Message("0718693002", "Test message");
        String result = invalidMsg.checkRecipientCell();
        assertEquals("Cell phone number is incorrectly formatted or does not contain an international code. Please correct the number and try again.", result);
    }
    
    @Test
    @Order(6)
    void testInvalidCellNumberTooShort() {
        Message invalidMsg = new Message("+27", "Test message");
        String result = invalidMsg.checkRecipientCell();
        assertEquals("Cell phone number is incorrectly formatted or does not contain an international code. Please correct the number and try again.", result);
    }
    
    @Test
    @Order(7)
    void testInvalidCellNumberContainsLetters() {
        Message invalidMsg = new Message("+27ABC123456", "Test message");
        String result = invalidMsg.checkRecipientCell();
        assertEquals("Cell phone number is incorrectly formatted or does not contain an international code. Please correct the number and try again.", result);
    }
    
    // ==================== TEST 3: Message Hash Creation ====================
    
    @Test
    @Order(8)
    void testMessageHashFormatForTestCase1() {
        // Test with provided test data
        Message testMsg = new Message("+27718693002", "Hi Mike, can you join us for dinner tonight?");
        String hash = testMsg.getMessageHash();
        
        // Format: first two numbers of message ID:message number:first and last words in caps
        String messageID = testMsg.getMessageID();
        String expectedFirstTwo = messageID.substring(0, 2);
        
        assertTrue(hash.startsWith(expectedFirstTwo + ":1:"));
        assertTrue(hash.endsWith("HIMIKE"));
        assertTrue(hash.contains(":"));
    }
    
    @Test
    @Order(9)
    void testMessageHashIsUpperCase() {
        Message testMsg = new Message("+447911123456", "Hello world testing");
        String hash = testMsg.getMessageHash();
        assertEquals(hash.toUpperCase(), hash, "Message hash should be in all caps");
    }
    
    @Test
    @Order(10)
    void testMessageHashContainsCorrectFormat() {
        Message testMsg = new Message("+27718693002", "Hello world");
        String hash = testMsg.getMessageHash();
        
        // Should contain two colons
        assertEquals(2, hash.length() - hash.replace(":", "").length(), 
                    "Hash should contain exactly 2 colons");
    }
    
    // ==================== TEST 4: Send/Store/Disregard Options ====================
    
    @Test
    @Order(11)
    void testSendMessageOption() {
        Message testMsg = new Message("+27718693002", "Test send message");
        String result = testMsg.sentMessage(1);
        
        assertTrue(result.contains("Message successfully sent"));
        assertTrue(result.contains("Press 0 to delete the message"));
        assertTrue(result.contains("Message successfully stored"));
        assertEquals(1, testMsg.getTotalMessagesSent());
    }
    
    @Test
    @Order(12)
    void testStoreMessageOption() {
        Message testMsg = new Message("+27718693002", "Test store message");
        String result = testMsg.sentMessage(2);
        
        assertEquals("Message successfully stored.", result);
        assertEquals(0, testMsg.getTotalMessagesSent());
        assertEquals(1, testMsg.getTotalMessagesStored());
    }
    
    @Test
    @Order(13)
    void testDisregardMessageOption() {
        Message testMsg = new Message("+27718693002", "Test disregard message");
        String result = testMsg.sentMessage(0);
        
        assertEquals("Press 0 to delete the message.", result);
    }
    
    // ==================== TEST 5: Message Length Validation ====================
    
    @Test
    @Order(14)
    void testMessageUnder250CharactersSuccess() {
        String shortMessage = "This is a short message under 250 characters.";
        String result = Message.validateMessageLength(shortMessage);
        assertEquals("Message ready to send.", result);
    }
    
    @Test
    @Order(15)
    void testMessageExactly250Characters() {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 250; i++) {
            sb.append("a");
        }
        String exactMessage = sb.toString();
        String result = Message.validateMessageLength(exactMessage);
        assertEquals("Message ready to send.", result);
    }
    
    @Test
    @Order(16)
    void testMessageExceeds250Characters() {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 260; i++) {
            sb.append("a");
        }
        String longMessage = sb.toString();
        String result = Message.validateMessageLength(longMessage);
        assertEquals("Message exceeds 250 characters by 10; please reduce the size.", result);
    }
    
    @Test
    @Order(17)
    void testMessageExceedsBySpecificAmount() {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 300; i++) {
            sb.append("b");
        }
        String longMessage = sb.toString();
        String result = Message.validateMessageLength(longMessage);
        assertEquals("Message exceeds 250 characters by 50; please reduce the size.", result);
    }
    
    // ==================== ADDITIONAL TESTS ====================
    
    @Test
    @Order(18)
    void testPrintMessages() {
        Message msg1 = new Message("+27718693002", "Hello there");
        Message msg2 = new Message("+27718693002", "How are you?");
        
        msg1.sentMessage(1); // Send
        msg2.sentMessage(2); // Store
        
        String output = msg1.printMessages();
        
        assertTrue(output.contains("SENT MESSAGES"));
        assertTrue(output.contains("STORED MESSAGES"));
        assertTrue(output.contains("Hello there"));
        assertTrue(output.contains("How are you?"));
    }
    
    @Test
    @Order(19)
    void testDeleteMessageByID() {
        Message testMsg = new Message("+27718693002", "Delete me");
        testMsg.sentMessage(1); // Send the message
        
        String messageID = testMsg.getMessageID();
        boolean deleted = Message.deleteMessage(messageID);
        
        assertTrue(deleted, "Message should be deleted successfully");
        assertEquals(0, testMsg.getTotalMessagesSent());
    }
    
    @Test
    @Order(20)
    void testMessageCounterIncrements() {
        Message msg1 = new Message("+27718693002", "Message 1");
        Message msg2 = new Message("+27718693002", "Message 2");
        Message msg3 = new Message("+27718693002", "Message 3");
        
        assertEquals(3, msg3.getNumMessagesSent());
        assertEquals(1, msg1.getNumMessagesSent());
        assertEquals(2, msg2.getNumMessagesSent());
    }
}
