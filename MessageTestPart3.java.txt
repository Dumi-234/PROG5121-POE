    // ==================== TEST 5: Delete Message Using Message Hash ====================
    
    @Test
    @Order(14)
    void testDeleteMessageByValidHash() {
        // Get a message hash from stored messages
        List<Message> storedMessages = Message.getStoredMessages();
        String messageHash = storedMessages.get(0).getMessageHash();
        String messageContent = storedMessages.get(0).getMessage();
        
        String result = Message.deleteMessageByHash(messageHash);
        
        assertEquals("Message: \"" + messageContent + "\" successfully deleted.", result);
        
        // Verify message is actually deleted
        List<Message> updatedStoredMessages = Message.getStoredMessages();
        assertEquals(1, updatedStoredMessages.size(), "Should have 1 stored message left");
    }

    @Test
    @Order(16)
    void testDeleteMessageByHashFromSentMessages() {
        // Get a message hash from sent messages
        List<Message> sentMessages = Message.getSentMessages();

        String messageContent = sentMessages.get(0).getMessage();

    // ==================== TEST 8: Delete Message Functionality ====================
    
    @Test
    @Order(26)
    void testDeleteMessageByHashThenVerifyRemovedFromAllArrays() {
        // Get a stored message's hash
        List<Message> storedMessages = Message.getStoredMessages();
        String messageHash = storedMessages.get(0).getMessageHash();
        
        // Get original sizes
        int originalStoredSize = Message.getStoredMessages().size();
        int originalHashesSize = Message.getMessageHashes().size();
        
        // Delete the message
        Message.deleteMessageByHash(messageHash);
        
        // Verify removed from stored messages
        assertEquals(originalStoredSize - 1, Message.getStoredMessages().size());
        
        // Verify removed from hashes array
        assertEquals(originalHashesSize - 1, Message.getMessageHashes().size());
        
        // Verify the hash is no longer in the list
        boolean hashStillExists = false;
        for (String hash : Message.getMessageHashes()) {
            if (hash.equals(messageHash)) {
                hashStillExists = true;
                break;
            }
        }
        assertFalse(hashStillExists, "Hash should be removed from messageHashes array");
    }
